#3.1
import matplotlib.cm as cm
from matplotlib.collections import LineCollection
def F(M1,m2,x1,y1,x2,y2):
    r=np.sqrt((x2-x1)**2+(y2-y1)**2)
    r=np.where(r==Re,np.nan,r)
    Ftot=-G*M1*m2/r**2
    Ftot=np.where(r<Re,np.nan,Ftot)
    Fx=Ftot*(x2-x1)/r
    Fy=Ftot*(y2-y1)/r
    return Fx,Fy

from scipy.integrate import quad
def dv(t):
    mt=m0-m*t
    mt=np.where(mt<mf,mf,mt)
    dv_more_t=ve*np.log(m0/mf)-g*T-g*(t-T)
    dv_t=np.where(t>T,dv_more_t,ve*np.log(m0/(mt))-g*t)

    return dv_t

height,error=quad(dv,0,T)
print('Altitude of rocket after time T: 'f'{height:.3g}','m')
print("Uncertainty of rocket's altitude: "f"{error:1g}","m")
